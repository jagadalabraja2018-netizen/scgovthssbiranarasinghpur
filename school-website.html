About Greenfield Academy

Rooted in tradition, inspired by the future — a journey of 40 years of educational excellence.

Founded in 1985, Greenfield Academy has been a pillar of quality education in the region. Our CBSE-affiliated institution serves students from Class 1 through Class 12 across Science, Commerce, and Arts streams.

Our sprawling 15-acre campus features state-of-the-art laboratories, a digital library, sports complex, auditorium, and dedicated spaces for arts and extracurricular activities. We believe holistic education shapes not just scholars, but well-rounded individuals.

With a student-to-teacher ratio of 20:1, our experienced faculty ensures personalized attention and guidance for every learner. Our consistent board exam results and placement record speak to the dedication of our community.

⚙️ Google Sheets Configuration

Enter your published Google Sheet CSV URLs below. Each sheet tab should be published as CSV from File → Share → Publish to web → CSV.

📜 Google Apps Script — Copy & Deploy

Go to extensions.google.com → Apps Script in your Google Sheet and paste this code. Then deploy as a Web App with access set to Anyone.

// ===== GREENFIELD ACADEMY - GOOGLE APPS SCRIPT =====
// Paste this in Google Apps Script and deploy as Web App

const SHEET_ID = 'YOUR_SPREADSHEET_ID'; // Replace with your Sheet ID

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheetName = data.sheet || 'Admissions';
    const sheet = ss.getSheetByName(sheetName);
    
    if (!sheet) {
      return ContentService.createTextOutput(
        JSON.stringify({ status: 'error', msg: 'Sheet not found' })
      ).setMimeType(ContentService.MimeType.JSON);
    }
    
    // Append row based on sheet type
    if (sheetName === 'Admissions') {
      sheet.appendRow([
        new Date().toLocaleDateString(),
        data.name, data.dob, data.class,
        data.gender, data.father, data.mother,
        data.phone, data.email, data.address,
        data.prevSchool, 'Pending'
      ]);
    } else if (sheetName === 'Contact') {
      sheet.appendRow([
        new Date().toLocaleString(),
        data.name, data.email,
        data.subject, data.message
      ]);
    }
    
    return ContentService.createTextOutput(
      JSON.stringify({ status: 'success', msg: 'Data saved!' })
    ).setMimeType(ContentService.MimeType.JSON);
    
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({ status: 'error', msg: err.toString() })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  // Health check endpoint
  return ContentService.createTextOutput(
    JSON.stringify({ status: 'ok', service: 'Greenfield Academy API' })
  ).setMimeType(ContentService.MimeType.JSON);
}
